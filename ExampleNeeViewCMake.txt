# Link to your NeeView target
target_link_libraries(neeview_core PRIVATE Anime4KCPP::Core)

# Optional: For runtime library paths (Linux)
if(NOT WIN32)
    set_target_properties(neeview_core PROPERTIES
        INSTALL_RPATH "$ORIGIN/../lib:${CMAKE_INSTALL_PREFIX}/lib"
    )
endif()

# Optional: Copy the library to your output directory during build
if(WIN32)
    add_custom_command(TARGET neeview_core POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${CMAKE_CURRENT_SOURCE_DIR}/../Anime4KCPP/install_shared/bin/ac.dll"
            "$<TARGET_FILE_DIR:neeview_core>/ac.dll"
        COMMENT "Copying Anime4KCPP DLL..."
    )
else()
    add_custom_command(TARGET neeview_core POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${CMAKE_CURRENT_SOURCE_DIR}/../Anime4KCPP/install_shared/lib/libac.so"
            "$<TARGET_FILE_DIR:neeview_core>/libac.so"
        COMMENT "Copying Anime4KCPP library..."
    )
endif()
